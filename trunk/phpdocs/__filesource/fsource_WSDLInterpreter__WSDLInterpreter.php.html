  146.     private function _validateNamingConvention($name)

  149.             preg_replace('#^[^a-zA-Z_\x7f-\xff]*#', '', $name));

  181.     /**
  182.      * Validates a wsdl type against known PHP primitive types, or otherwise
  183.      * validates the namespace of the type to PHP naming conventions
  184.      * 
  185.      * @param string $type the type to test
  186.      * 
  187.      * @return string the validated version of the submitted type
  188.      *

  191.      */    
  192.     private function _validateType($type) 
  193.     {
  194.         $array = false;
  195.         if (substr($type, -2) == "[]") {
  196.             $array = true;
  197.             $type = substr($type, 0, -2);
  198.         }
  199.         switch (strtolower($type)) {
  200.         case "int": case "integer": case "long": case "byte": case "short":
  201.         case "negativeInteger": case "nonNegativeInteger": 
  202.         case "nonPositiveInteger": case "positiveInteger":
  203.         case "unsignedByte": case "unsignedInt": case "unsignedLong": case "unsignedShort":
  204.             $validType = "integer";
  205.             break;
  206.             
  207.         case "float": case "long": case "double": case "decimal":
  208.             $validType = "double";
  209.             break;
  210.             
  211.         case "string": case "token": case "normalizedString": case "hexBinary":
  212.             $validType = "string";
  213.             break;
  214.             
  215.         default:
  216.             $validType = $this->_validateNamingConvention($type);
  217.             break;
  218.         }
  219.         if ($array) {
  220.             $validType .= "[]";
  221.         }
  222.         return $validType;
  223.     }

  254.     /**
  255.      * Generates the PHP code for a WSDL message type class representation
  256.      * 
  257.      * This gets a little bit fancy as the magic methods __get and __set in
  258.      * the generated classes are used for properties that are not named
  259.      * according to PHP naming conventions (e.g., "MY-VARIABLE").  These
  260.      * variables are set directly by SoapClient within the target class,
  261.      * and could normally be retrieved by $myClass->{"MY-VARIABLE"}.  For

  264.      * @param DOMElement $class the interpreted WSDL message type node
  265.      * @return string the php source code for the message type class
  266.      * 
  267.      * @access private
  268.      * @todo Include any applicable annotation from WSDL
  269.      */
  270.     private function _generateClassPHP($class) 
  271.     {
  272.         $return = "";
  273.         $return .= 'if (!class_exists("'.$class->getAttribute("validatedName").'")) {'."\n";

  275.         $return .= ' * '.$class->getAttribute("validatedName")."\n";
  276.         $return .= ' */'."\n";

  280.             $return .= " extends ".$extends->item(0)->nodeValue;

  283.     
  284.         $properties = $class->getElementsByTagName("property");
  285.         foreach ($properties as $property) {
  286.             $return .= "\t/**\n"
  287.                      . "\t * @access public\n"

  289.                      . "\t */\n"

  291.         }
  292.     
  293.         $extraParams = false;
  294.         $paramMapReturn = "\t".'private $_parameterMap = array ('."\n";
  295.         $properties = $class->getElementsByTagName("property");
  296.         foreach ($properties as $property) {
  297.             if ($property->getAttribute("name") != $property->getAttribute("validatedName")) {
  298.                 $extraParams = true;
  299.                 $paramMapReturn .= "\t\t".'"'.$property->getAttribute("name").
  300.                     '" => "'.$property->getAttribute("validatedName").'",'."\n";
  301.             }
  302.         }
  303.         $paramMapReturn .= "\t".');'."\n";
  304.         $paramMapReturn .= "\t".'/**'."\n";
  305.         $paramMapReturn .= "\t".' * Provided for setting non-php-standard named variables'."\n";

  317.             '{ return $this->{$this->_parameterMap[$var]}; }'."\n";
  318.         
  319.         if ($extraParams) {
  320.             $return .= $paramMapReturn;

  405.         $return .= "\t\t".'    }'."\n";
